CREATE TABLE DOMICILIO (
ID_DOMICILIO numeric ,
ID_CLIENTE numeric ,
ID_USUARIO numeric ,
FECHA_DOMICILIO datetime ,
HORA_DOMICILIO datetime ,
FECHA_VERIFICACION datetime ,
HORA_VERIFICACION datetime ,
DIRECCION varchar (300),
TELEFONO varchar (100),
BARRIO varchar (300),
ESTADO varchar (1),
OBSERVACIONES varchar (8000),
TOTAL numeric ,
CONSTRAINT PRIMARY KEY(ID_DOMICILIO)
)
