CREATE TABLE VIEW_PRODUCTO (
ID_PRODUCTO numeric ,
ID_IMAGEN numeric ,
ID_TIPO_PRODUCTO numeric ,
ID_EMPRESA numeric ,
NOM_PRODUCTO varchar (100),
PRECIO numeric ,
DESCRIPCION varchar (8000),
IMAGEN varchar (2147483647),
NOM_TIPO_PRODUCTO varchar (100),
NIT varchar (20),
RAZON_SOCIAL varchar (500),
SUCURSAL varchar (100),
CONSTRAINT PRIMARY KEY(ID_PRODUCTO)
)
